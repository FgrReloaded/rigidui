## Props

#### DraggableDashboard Props [!toc]

<TypeTable
  type={{
    children: {
      description:
        "The draggable wrapper components to be rendered in the dashboard",
      type: "ReactNode",
      required: true,
      default: "undefined",
    },
    className: {
      description: "Additional class names for styling the dashboard container",
      type: "string",
      default: "''",
    },
    showLockToggle: {
      description: "Whether to show the lock/unlock toggle switch",
      type: "boolean",
      default: "true",
    },
    showHandles: {
      description: "Whether to show drag handles on items",
      type: "boolean",
      default: "true",
    },
    gridCols: {
      description: "Number of grid columns for the dashboard layout",
      type: "number",
      default: "3",
    },
    gap: {
      description: "Gap between grid items (in Tailwind spacing units)",
      type: "number",
      default: "6",
    },
    defaultLocked: {
      description: "Whether the dashboard should be locked by default",
      type: "boolean",
      default: "false",
    },
    onOrderChange: {
      description: "Callback function called when item order changes",
      type: "(newOrder: string[]) => void",
      default: "undefined",
    },
    persistenceKey: {
      description:
        "Unique key for localStorage to persist item order across sessions",
      type: "string",
      default: "'draggable-dashboard-order'",
    },
  }}
/>
