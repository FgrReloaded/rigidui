## Features

<p className="md:text-sm mb-4!  text-gray-700 dark:text-gray-300 max-w-3xl">
  The Notification Center component is packed with powerful features designed to
  provide an exceptional user experience.
</p>

<Cards>
  <Card title="Dual Variants">
    Choose between full variant for comprehensive displays or popover variant
    for compact navigation integration.
  </Card>
  <Card title="Read Status Management">
    Track and manage notification read status with visual indicators and bulk
    actions.
  </Card>
  <Card title="Priority System">
    Organize notifications with high, medium, and low priority levels with
    visual color coding.
  </Card>
  <Card title="Real-time Updates">
    Optional real-time updates with configurable intervals to keep notifications
    fresh and current.
  </Card>
  <Card title="Filtering Options">
    Filter between all notifications or unread-only to help users focus on what
    matters most.
  </Card>
  <Card title="Notification Management">
    Complete CRUD operations with mark as read, mark all as read, and delete
    functionality.
  </Card>
  <Card title="Browser Notifications">
    Opt-in to receive native browser notifications for new updates, keeping you
    informed even when the app is in the background.
  </Card>
</Cards>

## Props

<TypeTable
  type={{
    className: {
      description: "Additional class names for styling",
      type: "string",
      default: "''",
    },
    variant: {
      description:
        "The display variant of the component - full for complete view or popover for compact trigger",
      type: "'full' | 'popover'",
      default: "'full'",
    },
    notifications: {
      description:
        "Static notifications array. If provided, will be used instead of fetchNotifications",
      type: "Notification[]",
      default: "undefined",
    },
    fetchNotifications: {
      description: "Function to fetch notifications asynchronously",
      type: "() => Promise<Notification[]>",
      default: "defaultFetchNotifications",
    },
    onMarkAsRead: {
      description:
        "Callback function called when a notification is marked as read",
      type: "(id: string) => Promise<void>",
      default: "defaultMarkAsRead",
    },
    onMarkAllAsRead: {
      description:
        "Callback function called when all notifications are marked as read",
      type: "() => Promise<void>",
      default: "defaultMarkAllAsRead",
    },
    onDeleteNotification: {
      description: "Callback function called when a notification is deleted",
      type: "(id: string) => Promise<void>",
      default: "defaultDeleteNotification",
    },
    onNotificationClick: {
      description: "Callback function called when a notification is clicked",
      type: "(notification: Notification) => void",
      default: "undefined",
    },
    enableBrowserNotifications: {
      description:
        "If true, the component will ask for permission and send browser notifications for new updates.",
      type: "boolean",
      default: "false",
    },
    maxHeight: {
      description: "Maximum height class for the notification list container",
      type: "string",
      default: "'h-96'",
    },
    showFilter: {
      description: "Whether to show the filter toggle button",
      type: "boolean",
      default: "true",
    },
    showMarkAllRead: {
      description: "Whether to show the mark all as read button",
      type: "boolean",
      default: "true",
    },
    enableRealTimeUpdates: {
      description: "Whether to enable automatic real-time updates",
      type: "boolean",
      default: "false",
    },
    updateInterval: {
      description:
        "Interval in milliseconds for real-time updates (when enabled)",
      type: "number",
      default: "30000",
    },
    emptyState: {
      description: "Custom empty state configuration",
      type: "{ title?: string; description?: string }",
      default:
        '{ title: "No notifications", description: "New notifications will appear here." }',
    },
  }}
